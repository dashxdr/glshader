#ifdef GL_ES
precision mediump float;
#endif

uniform float time;
uniform vec2 mouse;
uniform vec2 resolution;

/*
 * inspired by http://www.fractalforums.com/new-theories-and-research/very-simple-formula-for-fractal-patterns/
 * a slight(?) different 
 * public domain
 */
//dashxdr was here...

#define N 60
void main( void ) {
	vec2 v = (gl_FragCoord.xy - resolution/2.0) / min(resolution.y,resolution.x) * 20.0;

	float rsum = 0.0;
	float pi2 = 3.1415926535 * 2.0;
	float a = (mouse.x-.5)*pi2;
	float C = cos(a);
	float S = sin(a);
	vec2 xaxis=vec2(C, -S);
	vec2 yaxis=vec2(S, C);

	vec2 shift = vec2( 0, 1.618);
	float zoom = 1.0 + mouse.y*8.0;
	float decay=a>0.0 ? 1.01 : .99;
	for ( int i = 0; i < N; i++ ){
		float rr = dot(v,v);
		if ( rr > 0.618 ){
			rr = 1.618/rr ;
			v.x = v.x * rr;
			v.y = v.y * rr;
		}
		rsum *= decay;
		rsum += rr;
		
		v = vec2( dot(v,xaxis), dot(v,yaxis)) * zoom + shift;
	}
	
	float col = rsum * 0.5;


	gl_FragColor = vec4( cos(col*1.0), cos(col*1.618), cos(col*0.618), 1.0 );

}
